26// Network byte order conversion functions
27// Provide portable byte-order helpers with names that avoid clashes with system macros (htons/ntohs)
28namespace detail {
29 constexpr std::uint16_t bswap16(std::uint16_t x) noexcept { return static_cast<std::uint16_t>((x << 8) | (x >> 8)); }
30 constexpr std::uint32_t bswap32(std::uint32_t x) noexcept { return (x << 24) | ((x & 0x0000FF00U) << 8) | ((x & 0x00FF0000U) >> 8) | (x >> 24); }
31
32 // Compile-time endianness detection (fallback assumes little-endian for common targets)
33#if defined(__BYTE_ORDER__) && defined(__ORDER_LITTLE_ENDIAN__) && defined(__ORDER_BIG_ENDIAN__)
34 constexpr bool is_little_endian = (__BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__);
35#else
36 constexpr bool is_little_endian = true;
37#endif
38
39 // Convert between host and big-endian (network) order without colliding with system macros
40 constexpr std::uint16_t host_to_be16(std::uint16_t x) noexcept {
41 return is_little_endian ? bswap16(x) : x;
42 }
43 constexpr std::uint16_t be16_to_host(std::uint16_t x) noexcept {
44 return is_little_endian ? bswap16(x) : x;
45 }
46 constexpr std::uint32_t host_to_be32(std::uint32_t x) noexcept {
47 return is_little_endian ? bswap32(x) : x;
48 }
49 constexpr std::uint32_t be32_to_host(std::uint32_t x) noexcept {
50 return is_little_endian ? bswap32(x) : x;
51 }
52}

116struct CommonHeader {
117 // Byte 0: Transport specific (4 bits) + Message type (4 bits)
118 std::uint8_t transport_messageType;
119
120 // Byte 1: Reserved (4 bits) + Version (4 bits)
121 std::uint8_t reserved_version;
122
123 // Bytes 2-3: Message length (network byte order)
124 std::uint16_t messageLength;
125
126 // Byte 4: Domain number
127 std::uint8_t domainNumber;
128
129 // Byte 5: Minor version PTP
130 std::uint8_t minorVersionPTP;
131
132 // Bytes 6-7: Flags (network byte order)
133 std::uint16_t flagField;
134
135 // Bytes 8-15: Correction field (network byte order)
136 CorrectionField correctionField;
137
138 // Bytes 16-19: Message type specific (reserved)
139 std::uint32_t messageTypeSpecific;
140
141 // Bytes 20-29: Source port identity
142 PortIdentity sourcePortIdentity;
143
144 // Bytes 30-31: Sequence ID (network byte order)
145 std::uint16_t sequenceId;
146
147 // Byte 32: Control field (deprecated in v2, set to 0xFF)
148 std::uint8_t controlField;
149
150 // Byte 33: Mean log message interval
151 std::int8_t logMessageInterval;

153 //--------------------------------------------------------------------------
154 // Time-sensitive design accessors with compile-time optimization
155 //--------------------------------------------------------------------------
156
162 constexpr MessageType getMessageType() const noexcept {
163 return static_cast<MessageType>(transport_messageType & 0x0F);
164 }
165
171 constexpr void setMessageType(MessageType type) noexcept {
172 transport_messageType = (transport_messageType & 0xF0) |
173 (static_cast<std::uint8_t>(type) & 0x0F);
174 }
175
180 constexpr std::uint8_t getVersion() const noexcept {
181 return reserved_version & 0x0F;
182 }
183
188 constexpr void setVersion(std::uint8_t version) noexcept {
189 reserved_version = (reserved_version & 0xF0) | (version & 0x0F);
190 }
191
196 inline bool isTwoStep() const noexcept {
197 return (detail::be16_to_host(flagField) & Flags::TWO_STEP) != 0;
198 }
199
205 constexpr bool isEventMessage() const noexcept {
206 const auto type = getMessageType();
207 return (type == MessageType::Sync) ||
208 (type == MessageType::Delay_Req) ||
209 (type == MessageType::Pdelay_Req) ||
210 (type == MessageType::Pdelay_Resp);
211 }
212
218 PTPResult<void> validate() const noexcept {
219 // Version check
220 if (getVersion() != 2) {
221 return PTPResult<void>::makeError(PTPError::INVALID_VERSION);
222 }
223
224 // Message length bounds check
225 const auto msgLen = detail::be16_to_host(messageLength);
226 if (msgLen < sizeof(CommonHeader) || msgLen > 1500) {
227 return PTPResult<void>::makeError(PTPError::INVALID_LENGTH);
228 }
229
230 // Reserved fields should be zero
231 if ((reserved_version & 0xF0) != 0) {
232 return PTPResult<void>::makeError(PTPError::INVALID_RESERVED_FIELD);
233 }
234
235 return PTPResult<void>::success();
236 }
237};

518template<typename BodyType>
519struct PTPMessage {
520 CommonHeader header;
521 BodyType body;
522
528 PTPResult<void> validate() const noexcept {
529 auto headerResult = header.validate();
530 if (!headerResult.isSuccess()) {
531 return headerResult;
532 }
533
534 return body.validate();
535 }
536
542 static constexpr std::size_t getMessageSize() noexcept {
543 return sizeof(CommonHeader) + sizeof(BodyType);
544 }
545
552 void initialize(MessageType msgType, std::uint8_t domain,
553 const PortIdentity& sourcePort) noexcept {
554 // Initialize header with IEEE 1588-2019 defaults
555 header = {};
556 header.setMessageType(msgType);
557 header.setVersion(2); // IEEE 1588-2019 is version 2
558 header.messageLength = detail::host_to_be16(static_cast<std::uint16_t>(getMessageSize()));
559 header.domainNumber = domain;
560 header.minorVersionPTP = 1; // IEEE 1588-2019 minor version
561 header.sourcePortIdentity = sourcePort;
562 header.controlField = 0xFF; // Deprecated in v2
563
564 // Initialize body to zero
565 body = {};
566 }
567};

705 const std::uint8_t action = getActionField();
