68inline bool consume_bmca_tie_token() noexcept {
69 int cur = detail::bmca_tie_tokens().load(std::memory_order_relaxed);
70 while (cur > 0) {
71 if (detail::bmca_tie_tokens().compare_exchange_weak(cur, cur - 1, std::memory_order_relaxed)) {
72 return true;
73 }

75 }
