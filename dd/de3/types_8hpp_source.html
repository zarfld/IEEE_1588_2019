292struct TimeInterval {
293 Integer64 scaled_nanoseconds;
294
299 constexpr double toNanoseconds() const noexcept {
300 return static_cast<double>(scaled_nanoseconds) / 65536.0;
301 }
302
307 static constexpr TimeInterval fromNanoseconds(double ns) noexcept {
308 return {static_cast<Integer64>(ns * 65536.0)};
309 }
310
317 static constexpr TimeInterval fromNanosecondsUnbiased(double ns) noexcept {
318 // Convert to scaled double
319 double scaled = ns * 65536.0;
320 // Extract integer part
321 Integer64 base = static_cast<Integer64>(scaled);
322 double frac = scaled - static_cast<double>(base);
323 if (frac > 0.5) {
324 return {base + 1};
325 } else if (frac < 0.5) {
326 return {base};
327 }
328 // frac == 0.5 within double precision: round to even
329 return { (base & 1) == 0 ? base : base + 1 };
330 }
331};

348inline TimeInterval Timestamp::operator-(const Timestamp& other) const noexcept {
349 // Calculate difference in nanoseconds, then convert to TimeInterval
350 UInteger64 this_ns = getTotalSeconds() * 1000000000ULL + nanoseconds;
351 UInteger64 other_ns = other.getTotalSeconds() * 1000000000ULL + other.nanoseconds;
352 Integer64 diff_ns = static_cast<Integer64>(this_ns) - static_cast<Integer64>(other_ns);
353 return TimeInterval{diff_ns * 65536}; // Convert to scaled nanoseconds (2^16 factor)
354}

356inline Timestamp Timestamp::operator*(std::uint8_t multiplier) const noexcept {
357 UInteger64 total_ns = getTotalSeconds() * 1000000000ULL + nanoseconds;
358 total_ns *= multiplier;
359
360 Timestamp result;
361 result.setTotalSeconds(total_ns / 1000000000ULL);
362 result.nanoseconds = static_cast<UInteger32>(total_ns % 1000000000ULL);
363 return result;
364}
