321 Types::PTPResult<Types::TimeInterval> calculateOffset(
322 const Types::Timestamp& sync_timestamp,
323 const Types::Timestamp& sync_reception,
324 const Types::Timestamp& delay_req_timestamp,
325 const Types::Timestamp& delay_resp_timestamp
326 ) noexcept {
327 // Ordering assertions (FM-001): T2>=T1 and T4>=T3 must hold
328 if (sync_reception < sync_timestamp) {
329 Common::utils::logging::warn("Timestamps", 0x0204, "Sync RX timestamp earlier than origin (T2 < T1)");
330 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
331 Common::utils::health::emit();
332 }
333 if (delay_resp_timestamp < delay_req_timestamp) {
334 Common::utils::logging::warn("Timestamps", 0x0205, "Delay response RX earlier than request TX (T4 < T3)");
335 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
336 Common::utils::health::emit();
337 }
338 // IEEE 1588-2019 E2E algorithm:
339 // offset_from_master = ((T2 - T1) - (T4 - T3)) / 2
340 const Types::TimeInterval t2_minus_t1 = (sync_reception - sync_timestamp);
341 const Types::TimeInterval t4_minus_t3 = (delay_resp_timestamp - delay_req_timestamp);
342 // Work directly on scaled nanoseconds (2^-16 ns units) to avoid float rounding
343 const Types::Integer64 diff_scaled = (t2_minus_t1.scaled_nanoseconds - t4_minus_t3.scaled_nanoseconds);
344 Types::Integer64 scaled;
345 // Optional FM-014 mitigation: unbiased half-to-even division by 2 to remove rounding bias when odd values appear in scaled domain.
346 if (Common::utils::config::is_rounding_compensation_enabled()) {
347 // Banker's rounding for division by 2
348 const Types::Integer64 n = diff_scaled / 2; // trunc towards 0
349 const Types::Integer64 r = diff_scaled % 2; // remainder, same sign as numerator
350 if (r == 0) {
351 scaled = n;
352 } else {
353 // Tie at .5: round to even result
354 if ((n & 1) != 0) {
355 scaled = n + (diff_scaled > 0 ? 1 : -1);
356 } else {
357 scaled = n;
358 }
359 }
360 } else {
361 scaled = diff_scaled / 2;
362 }
363 Types::Integer64 adjusted = scaled;
364 if (Common::utils::fi::is_offset_jitter_enabled()) {
365 adjusted += (Common::utils::fi::get_offset_jitter_ns() << 16); // convert ns to scaled (2^-16 ns)
366 }
367 // Range validation & clamp (mitigation FM-002/FM-013)
368 constexpr Types::Integer64 MAX_ABS_SCALED = static_cast<Types::Integer64>(1ull << 46); // ~ 2^30 ns after division margin
369 if (adjusted > MAX_ABS_SCALED) {
370 adjusted = MAX_ABS_SCALED;
371 Common::utils::logging::warn("Offset", 0x0202, "Offset clamped positive upper bound");
372 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
373 } else if (adjusted < -MAX_ABS_SCALED) {
374 adjusted = -MAX_ABS_SCALED;
375 Common::utils::logging::warn("Offset", 0x0203, "Offset clamped negative lower bound");
376 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
377 } else {
378 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsPassed, 1);
379 }
380 offsetFromMaster = Types::TimeInterval{adjusted};
381 Common::utils::logging::debug("Offset", 0x0200, "Offset from master calculated");
382 Common::utils::metrics::increment(Common::utils::metrics::CounterId::OffsetsComputed, 1);
383 Common::utils::health::record_offset_ns(static_cast<long long>(offsetFromMaster.toNanoseconds()));
384 Common::utils::health::emit();
385 return Types::PTPResult<Types::TimeInterval>::success(offsetFromMaster);
386 }
