82void SyncIntegration::update_statistics(double offset_ns, double delay_ns) noexcept {
83 // Update offset statistics
84 statistics_.total_offset_samples++;
85 statistics_.current_offset_ns = offset_ns;
86
87 // Track min/max offset
88 if (statistics_.total_offset_samples == 1) {
89 statistics_.min_offset_ns = offset_ns;
90 statistics_.max_offset_ns = offset_ns;
91 statistics_.avg_offset_ns = offset_ns;
92 } else {
93 statistics_.min_offset_ns = std::min(statistics_.min_offset_ns, offset_ns);
94 statistics_.max_offset_ns = std::max(statistics_.max_offset_ns, offset_ns);
95
96 // Running average (incremental calculation)
97 double delta = offset_ns - statistics_.avg_offset_ns;
98 statistics_.avg_offset_ns += delta / statistics_.total_offset_samples;
99 }
100
101 // Track sub-microsecond accuracy
102 if (std::abs(offset_ns) < 1000.0) { // < 1µs
103 statistics_.sub_microsecond_samples++;
104 }
105
106 // Update delay statistics
107 if (delay_ns >= 0.0) { // Valid delay
108 statistics_.total_delay_samples++;
109 statistics_.current_delay_ns = delay_ns;
110
111 if (statistics_.total_delay_samples == 1) {
112 statistics_.min_delay_ns = delay_ns;
113 statistics_.max_delay_ns = delay_ns;
114 statistics_.avg_delay_ns = delay_ns;
115 } else {
116 statistics_.min_delay_ns = std::min(statistics_.min_delay_ns, delay_ns);
117 statistics_.max_delay_ns = std::max(statistics_.max_delay_ns, delay_ns);
118
119 double delta_delay = delay_ns - statistics_.avg_delay_ns;
120 statistics_.avg_delay_ns += delta_delay / statistics_.total_delay_samples;
121 }
122 } else {
123 statistics_.negative_delay_count++;
124 }
125
126 // Update variance calculation (rolling window)
127 offset_samples_.push_back(offset_ns);
128
129 // Keep only recent samples for variance
130 if (offset_samples_.size() > config_.variance_window_samples) {
131 offset_samples_.erase(offset_samples_.begin());
132 }
133
134 // Calculate variance if we have enough samples
135 if (offset_samples_.size() >= 2) {
136 calculate_variance();
137 }
138}

140void SyncIntegration::calculate_variance() noexcept {
141 // Calculate mean of samples
142 double sum = 0.0;
143 for (double sample : offset_samples_) {
144 sum += sample;
145 }
146 double mean = sum / offset_samples_.size();
147
148 // Calculate variance
149 double variance_sum = 0.0;
150 for (double sample : offset_samples_) {
151 double diff = sample - mean;
152 variance_sum += diff * diff;
153 }
154
155 statistics_.offset_variance_ns2 = variance_sum / offset_samples_.size();
156 statistics_.offset_std_dev_ns = std::sqrt(statistics_.offset_variance_ns2);
157}

159void SyncIntegration::update_health_status(const Types::Timestamp& current_time) noexcept {
160 // Update timestamp
161 health_.timestamp_ns = current_time.getTotalSeconds() * 1'000'000'000ULL + current_time.nanoseconds;
162
163 // Start with critical status, improve based on conditions
164 health_.status = SyncHealthStatus::Status::Critical;
165 health_.message.clear();
166
167 // Check if we have any samples
168 if (statistics_.total_offset_samples == 0) {
169 health_.message = "No synchronization samples yet";
170 health_.offset_within_spec = false;
171 health_.delay_stable = false;
172 health_.measurements_valid = false;
173 return;
174 }
175
176 // Check offset accuracy
177 double abs_offset = std::abs(statistics_.current_offset_ns);
178
179 if (abs_offset < config_.synchronized_threshold_ns) {
180 health_.offset_within_spec = true;
181 health_.status = SyncHealthStatus::Status::Synchronized;
182 health_.message = "Synchronized (< 1µs)";
183 } else if (abs_offset < config_.degraded_threshold_ns) {
184 health_.offset_within_spec = false;
185 health_.status = SyncHealthStatus::Status::Converging;
186 health_.message = "Converging (offset < 10µs)";
187 } else if (abs_offset < config_.critical_threshold_ns) {
188 health_.offset_within_spec = false;
189 health_.status = SyncHealthStatus::Status::Degraded;
190 health_.message = "Degraded (offset < 100µs)";
191 } else {
192 health_.offset_within_spec = false;
193 health_.status = SyncHealthStatus::Status::Critical;
194 health_.message = "Critical (offset > 100µs)";
195 }
196
197 // Check delay stability (low variance = stable)
198 if (statistics_.offset_std_dev_ns < 500.0) { // < 500ns std dev
199 health_.delay_stable = true;
200 } else {
201 health_.delay_stable = false;
202 if (health_.status == SyncHealthStatus::Status::Synchronized) {
203 health_.status = SyncHealthStatus::Status::Converging;
204 health_.message = "Converging (high variance)";
205 }
206 }
207
208 // Check measurement validity
209 double error_rate = 0.0;
210 if (statistics_.total_offset_samples > 0) {
211 error_rate = static_cast<double>(statistics_.negative_delay_count +
212 statistics_.timestamp_order_violations) /
213 statistics_.total_offset_samples;
214 }
215
216 if (error_rate < 0.01) { // < 1% error rate
217 health_.measurements_valid = true;
218 } else {
219 health_.measurements_valid = false;
220 if (health_.status != SyncHealthStatus::Status::Critical) {
221 health_.status = SyncHealthStatus::Status::Degraded;
222 health_.message = "Degraded (measurement errors)";
223 }
224 }
225
226 // TODO: Task 3 - Check servo lock status
227 health_.servo_locked = false; // Not yet implemented
