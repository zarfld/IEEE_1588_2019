54inline const char* servo_state_to_string(ServoState state) {
55 switch (state) {
56 case ServoState::Uninitialized: return "Uninitialized";
57 case ServoState::Unlocked: return "Unlocked";
58 case ServoState::Locking: return "Locking";
59 case ServoState::Locked: return "Locked";
60 case ServoState::Holdover: return "Holdover";
61 default: return "Unknown";
62 }
63}
