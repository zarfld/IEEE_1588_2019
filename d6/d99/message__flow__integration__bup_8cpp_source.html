97Types::PTPError MessageFlowCoordinator::process_announce_message(
98 const AnnounceMessage& message,
99 std::uint64_t reception_timestamp_ns
100) noexcept {
101 if (!is_running_) {
102 return Types::PTPError::NOT_INITIALIZED;
103 }
104
105 statistics_.announce_received++;
106
107 // Validate message header
108 auto header_result = validate_message_header(message.header);
109 if (header_result != Types::PTPError::Success) {
110 statistics_.announce_errors++;
111 statistics_.invalid_messages++;
112 return header_result;
113 }
114
115 // Check domain
116 auto domain_result = check_domain(message.header.domainNumber);
117 if (domain_result != Types::PTPError::Success) {
118 statistics_.announce_errors++;
119 statistics_.domain_mismatches++;
120 return domain_result;
121 }
122
123 // Check message age
124 auto age_result = check_message_age(
125 reception_timestamp_ns,
126 reception_timestamp_ns // Current time is reception time
127 );
128 if (age_result != Types::PTPError::Success) {
129 statistics_.announce_errors++;
130 return age_result;
131 }
132
133 // Update timing statistics
134 update_timing_statistics(true, reception_timestamp_ns);
135
136 // Trigger BMCA execution if enabled
137 if (config_.enable_bmca_on_announce) {
138 // The BMCA coordinator will handle the foreign master update
139 // and perform best master selection
140 statistics_.bmca_triggered++;
141
142 // Note: In a real implementation, we would extract the foreign master
143 // info from the Announce message and pass it to BMCA coordinator.
144 // For now, we rely on the BMCA coordinator's periodic execution.
145 }
146
147 statistics_.announce_processed++;
148
149 // Update health status
150 update_health_status();
151
152 return Types::PTPError::Success;
153}

275MessageFlowHealthStatus MessageFlowCoordinator::get_health_status() const noexcept {
276 MessageFlowHealthStatus health;
277
278 // Get current time for age calculations
279 const std::uint64_t current_time = statistics_.last_sync_time_ns;
280
281 // Check message flow activity
282 health.announce_flow_active = (statistics_.announce_received > 0);
283 health.sync_flow_active = (statistics_.sync_received > 0);
284
285 // Check BMCA operational
286 health.bmca_operational = (statistics_.bmca_triggered > 0);
287
288 // Check servo operational
289 health.servo_operational = (statistics_.servo_adjustments > 0);
290
291 // Check timing spec (message intervals within expected ranges)
292 const bool announce_timing_ok = (statistics_.announce_interval_ns > 0) &&
293 (statistics_.announce_interval_ns < config_.announce_timeout_ns);
294 const bool sync_timing_ok = (statistics_.sync_interval_ns > 0) &&
295 (statistics_.sync_interval_ns < config_.sync_timeout_ns);
296 health.within_timing_spec = announce_timing_ok && sync_timing_ok;
297
298 // Get component health (would query actual coordinators in real implementation)
299 health.bmca_healthy = bmca_.is_running();
300 health.sync_healthy = sync_.is_running();
301 health.servo_healthy = servo_.is_running();
302
303 // Determine overall status
304 const int healthy_components =
305 (health.bmca_healthy ? 1 : 0) +
306 (health.sync_healthy ? 1 : 0) +
307 (health.servo_healthy ? 1 : 0);
308
309 const bool has_errors = (statistics_.announce_errors > 0) ||
310 (statistics_.sync_errors > 0) ||
311 (statistics_.invalid_messages > 10);
312
313 if (healthy_components == 3 && !has_errors &&
314 health.announce_flow_active && health.sync_flow_active) {
315 health.status = MessageFlowHealthStatus::Status::Healthy;
316 health.message = "All message flows operational";
317 } else if (healthy_components >= 2 || !has_errors) {
318 health.status = MessageFlowHealthStatus::Status::Degraded;
319 health.message = "Some message flows experiencing issues";
320 } else {
321 health.status = MessageFlowHealthStatus::Status::Critical;
322 health.message = "Major message flow failures detected";
323 }
324
325 health.timestamp_ns = current_time;
326
327 return health;
328}

334Types::PTPError MessageFlowCoordinator::validate_message_header(
335 const CommonHeader& header
336) noexcept {
337 // Check PTP version (IEEE 1588-2019 is version 2)
338 const std::uint8_t version = header.getVersion();
339 if (version != 2) {
340 return Types::PTPError::INVALID_VERSION;
341 }
342
343 // Check message length is reasonable
344 const std::uint16_t length = detail::be16_to_host(header.messageLength);
345 if (length == 0 || length > 1500) { // Ethernet MTU limit
346 return Types::PTPError::INVALID_LENGTH;
347 }
348
349 return Types::PTPError::Success;
350}

386void MessageFlowCoordinator::update_timing_statistics(
387 bool is_announce,
388 std::uint64_t timestamp_ns
389) noexcept {
390 if (is_announce) {
391 // Calculate announce interval
392 if (!first_announce_ && statistics_.last_announce_time_ns > 0) {
393 const std::uint64_t interval = timestamp_ns - statistics_.last_announce_time_ns;
394
395 // Simple exponential moving average (alpha = 0.125 for smoothing)
396 if (statistics_.announce_interval_ns == 0) {
397 statistics_.announce_interval_ns = interval;
398 } else {
399 statistics_.announce_interval_ns =
400 (statistics_.announce_interval_ns * 7 + interval) / 8;
401 }
402 } else {
403 first_announce_ = false;
404 }
405
406 statistics_.last_announce_time_ns = timestamp_ns;
407 } else {
408 // Calculate sync interval
409 if (!first_sync_ && statistics_.last_sync_time_ns > 0) {
410 const std::uint64_t interval = timestamp_ns - statistics_.last_sync_time_ns;
411
412 // Simple exponential moving average
413 if (statistics_.sync_interval_ns == 0) {
414 statistics_.sync_interval_ns = interval;
415 } else {
416 statistics_.sync_interval_ns =
417 (statistics_.sync_interval_ns * 7 + interval) / 8;
418 }
419 } else {
420 first_sync_ = false;
421 }
422
423 statistics_.last_sync_time_ns = timestamp_ns;
424 }
