59 // For now generate a simple pattern derived from port number to avoid all-zero identity.

165Types::PTPResult<void> PtpPort::process_event(StateEvent event) noexcept {
166 PortState current_state = port_data_set_.port_state;
167 PortState new_state = current_state;
168
169 // State machine transitions per IEEE 1588-2019 Figure 9-1
170 switch (current_state) {
171 case PortState::Initializing:
172 switch (event) {
173 case StateEvent::INITIALIZE:
174 new_state = PortState::Listening;
175 break;
176 case StateEvent::FAULT_DETECTED:
177 new_state = PortState::Faulty;
178 break;
179 case StateEvent::DESIGNATED_DISABLED:
180 new_state = PortState::Disabled;
181 break;
182 default:
183 break;
184 }
185 break;
186
187 case PortState::Faulty:
188 switch (event) {
189 case StateEvent::FAULT_CLEARED:
190 new_state = PortState::Initializing;
191 break;
192 default:
193 break;
194 }
195 break;
196
197 case PortState::Disabled:
198 switch (event) {
199 case StateEvent::DESIGNATED_ENABLED:
200 new_state = PortState::Listening;
201 break;
202 default:
203 break;
204 }
205 break;
206
207 case PortState::Listening:
208 switch (event) {
209 case StateEvent::RS_MASTER:
210 case StateEvent::RS_GRAND_MASTER:
211 new_state = PortState::PreMaster;
212 break;
213 case StateEvent::RS_SLAVE:
214 new_state = PortState::Uncalibrated;
215 break;
216 case StateEvent::RS_PASSIVE:
217 new_state = PortState::Passive;
218 break;
219 case StateEvent::FAULT_DETECTED:
220 new_state = PortState::Faulty;
221 break;
222 case StateEvent::DESIGNATED_DISABLED:
223 new_state = PortState::Disabled;
224 break;
225 default:
226 break;
227 }
228 break;
229
230 case PortState::PreMaster:
231 switch (event) {
232 case StateEvent::QUALIFICATION_TIMEOUT:
233 new_state = PortState::Master;
234 break;
235 case StateEvent::RS_SLAVE:
236 new_state = PortState::Uncalibrated;
237 break;
238 case StateEvent::RS_PASSIVE:
239 new_state = PortState::Passive;
240 break;
241 default:
242 break;
243 }
244 break;
245
246 case PortState::Master:
247 switch (event) {
248 case StateEvent::RS_SLAVE:
249 new_state = PortState::Uncalibrated;
250 break;
251 case StateEvent::RS_PASSIVE:
252 new_state = PortState::Passive;
253 break;
254 default:
255 break;
256 }
257 break;
258
259 case PortState::Passive:
260 switch (event) {
261 case StateEvent::RS_MASTER:
262 case StateEvent::RS_GRAND_MASTER:
263 new_state = PortState::PreMaster;
264 break;
265 case StateEvent::RS_SLAVE:
266 new_state = PortState::Uncalibrated;
267 break;
268 default:
269 break;
270 }
271 break;
272
273 case PortState::Uncalibrated:
274 switch (event) {
275 case StateEvent::RS_MASTER:
276 case StateEvent::RS_GRAND_MASTER:
277 new_state = PortState::PreMaster;
278 break;
279 case StateEvent::RS_PASSIVE:
280 new_state = PortState::Passive;
281 break;
282 case StateEvent::SYNCHRONIZATION_FAULT:
283 new_state = PortState::Listening;
284 break;
285 case StateEvent::ANNOUNCE_RECEIPT_TIMEOUT:
286 new_state = PortState::Listening;
287 break;
288 default:
289 // Check if synchronization is achieved (implementation-specific)
290 // For now, transition to SLAVE after receiving valid Sync+Follow_Up
291 break;
292 }
293 break;
294
295 case PortState::Slave:
296 switch (event) {
297 case StateEvent::RS_MASTER:
298 case StateEvent::RS_GRAND_MASTER:
299 new_state = PortState::PreMaster;
300 break;
301 case StateEvent::RS_PASSIVE:
302 new_state = PortState::Passive;
303 break;
304 case StateEvent::SYNCHRONIZATION_FAULT:
305 new_state = PortState::Uncalibrated;
306 break;
307 case StateEvent::ANNOUNCE_RECEIPT_TIMEOUT:
308 new_state = PortState::Listening;
309 break;
310 default:
311 break;
312 }
313 break;
314 }
315
316 // Apply state transition if changed
317 if (new_state != current_state) {
318 return transition_to_state(new_state);
319 }
320
321 return Types::PTPResult<void>{};
322}

324Types::PTPResult<void> PtpPort::transition_to_state(PortState new_state) noexcept {
325 PortState old_state = port_data_set_.port_state;
326 port_data_set_.port_state = new_state;
327 statistics_.state_transitions++;
328 // Reset heuristic counter when entering UNCALIBRATED; clear on others as well
329 if (new_state == PortState::Uncalibrated) {
330 successful_offsets_in_window_ = 0;
331 } else if (new_state != old_state) {
332 successful_offsets_in_window_ = 0;
333 }
334
335 // State exit actions
336 switch (old_state) {
337 case PortState::Master:
338 // Stop transmitting Announce and Sync messages
339 break;
340 case PortState::Slave:
341 case PortState::Uncalibrated:
342 // Stop requesting delay measurements
343 break;
344 default:
345 break;
346 }
347
348 // State entry actions per IEEE 1588-2019 Section 9.2.5
349 switch (new_state) {
350 case PortState::Initializing:
351 // Clear foreign master list
352 foreign_master_count_ = 0;
353 break;
354
355 case PortState::Listening:
356 // Start listening for Announce messages
357 // Reset announce timeout
358 announce_timeout_time_ = Types::Timestamp{};
359 // Reset parent data set when returning to Listening from Slave/Uncalibrated (no master selected)
360 // Per IEEE 1588-2019 Section 9.2.6.11: Announce receipt timeout clears parent
361 // CRITICAL: Set GM identity to local clock identity (not zeros) to avoid BMCA issues
362 // Don't clear on initial transition from Initializing to avoid affecting initialization
363 if (old_state == PortState::Slave || old_state == PortState::Uncalibrated) {
364 parent_data_set_.parent_port_identity.port_number = 0;
365 parent_data_set_.parent_port_identity.clock_identity.fill(0);
366 parent_data_set_.parent_stats = false;
367 parent_data_set_.observed_parent_offset_scaled_log_variance = 0xFFFF;
368 parent_data_set_.observed_parent_clock_phase_change_rate = 0x7FFFFFFF;
369 // Set GM identity to local clock (not zeros) per constructor logic
370 parent_data_set_.grandmaster_identity = port_data_set_.port_identity.clock_identity;
371 parent_data_set_.grandmaster_clock_quality.clock_class = 248;
372 parent_data_set_.grandmaster_clock_quality.clock_accuracy = 0xFE;
373 parent_data_set_.grandmaster_clock_quality.offset_scaled_log_variance = 0xFFFF;
374 parent_data_set_.grandmaster_priority1 = 128;
375 parent_data_set_.grandmaster_priority2 = 128;
376 }
377 break;
378
379 case PortState::PreMaster:
380 // Start qualification timeout (implementation-specific duration)
381 // Typically 2 * announce interval
382 break;
383
384 case PortState::Master:
385 // Start transmitting Announce and Sync messages
386 last_announce_time_ = Types::Timestamp{};
387 last_sync_time_ = Types::Timestamp{};
388 break;
389
390 case PortState::Slave:
391 case PortState::Uncalibrated:
392 // Start requesting delay measurements
393 last_delay_req_time_ = Types::Timestamp{};
394 break;
395
396 case PortState::Faulty:
397 statistics_.fault_events++;
398 if (callbacks_.on_fault) {
399 callbacks_.on_fault("Port entered FAULTY state");
400 }
401 break;
402
403 default:
404 break;
405 }
406
407 // Notify state change
408 if (callbacks_.on_state_change) {
409 callbacks_.on_state_change(old_state, new_state);
410 }
411
412 return Types::PTPResult<void>{};
413}

415Types::PTPResult<void> PtpPort::process_announce(const AnnounceMessage& message) noexcept {
416 statistics_.announce_messages_received++;
417
418 // Update last announce time for timeout tracking (IEEE 1588-2019 Section 9.5.17)
419 // This is used by the announce receipt timeout mechanism in Slave/Uncalibrated states
420 last_announce_time_ = callbacks_.get_timestamp ? callbacks_.get_timestamp() : Types::Timestamp{};
421
422 // Update foreign master list
423 auto result = update_foreign_master_list(message);
424 if (!result.hasValue()) {
425 return result;
426 }
427
428 // Extract timePropertiesDS from Announce message per IEEE 1588-2019 Section 8.2.4
429 // Flag field bits (Section 13.3.2.6)
430 const std::uint16_t flags = detail::be16_to_host(message.header.flagField);
431 time_properties_data_set_.leap61 = (flags & Flags::LI_61) != 0;
432 time_properties_data_set_.leap59 = (flags & Flags::LI_59) != 0;
433 time_properties_data_set_.currentUtcOffsetValid = (flags & Flags::CURRENT_UTC_OFFSET_VALID) != 0;
434 time_properties_data_set_.ptpTimescale = (flags & Flags::PTP_TIMESCALE) != 0;
435 time_properties_data_set_.timeTraceable = (flags & Flags::TIME_TRACEABLE) != 0;
436 time_properties_data_set_.frequencyTraceable = (flags & Flags::FREQUENCY_TRACEABLE) != 0;
437
438 // Announce body fields (Section 13.5, Table 34)
439 time_properties_data_set_.currentUtcOffset = detail::be16_to_host(message.body.currentUtcOffset);
440 time_properties_data_set_.timeSource = message.body.timeSource;
441
442 // Run BMCA if in appropriate state
443 if (port_data_set_.port_state == PortState::Listening ||
444 port_data_set_.port_state == PortState::PreMaster ||
445 port_data_set_.port_state == PortState::Master ||
446 port_data_set_.port_state == PortState::Passive ||
447 port_data_set_.port_state == PortState::Uncalibrated ||
448 port_data_set_.port_state == PortState::Slave) {
449
450 return run_bmca();
451 }
452
453 return Types::PTPResult<void>{};

474Types::PTPResult<void> PtpPort::process_follow_up(const FollowUpMessage& message) noexcept {
475 statistics_.follow_up_messages_received++;
476
477 // Only process in slave states
478 if (port_data_set_.port_state != PortState::Uncalibrated &&
479 port_data_set_.port_state != PortState::Slave) {
480 return Types::PTPResult<void>::success();
481 }
482
483 sync_origin_timestamp_ = message.body.preciseOriginTimestamp; // T1
484 // Store correctionField from Follow_Up message per IEEE 1588-2019 Section 11.3.2
485 // CorrectionField is in units of 2^-16 nanoseconds (scaled nanoseconds)
486 follow_up_correction_ = Types::TimeInterval::fromNanoseconds(message.header.correctionField.toNanoseconds());
487 have_follow_up_ = true;
488 if (have_sync_ && have_delay_req_ && have_delay_resp_) {
489 auto result = calculate_offset_and_delay();
490 if (!result.is_success()) {
491 return result;
492 }
493 }
494
495 // Check if we can transition from UNCALIBRATED to SLAVE
496 if (port_data_set_.port_state == PortState::Uncalibrated) {
497 // Tightened sync heuristic (FM-008): require >=3 local successful offsets and zero global validation failures
498 constexpr std::uint32_t MIN_OFFSETS_FOR_SYNC = 3; // need 3 stable samples
499 auto fails = Common::utils::metrics::get(Common::utils::metrics::CounterId::ValidationsFailed);
500 if (successful_offsets_in_window_ >= MIN_OFFSETS_FOR_SYNC && fails == 0) {
501 Common::utils::logging::info("Heuristic", 0x0401, "Transition to SLAVE after stable offset samples");
502 return transition_to_state(PortState::Slave);
503 } else {
504 Common::utils::logging::debug("Heuristic", 0x0400, "Remaining UNCALIBRATED: samples insufficient or validation failures present");
505 }
506 }
507
508 return Types::PTPResult<void>::success();

511Types::PTPResult<void> PtpPort::process_delay_req(const DelayReqMessage& message,
512 const Types::Timestamp& rx_timestamp) noexcept {
513 statistics_.delay_req_messages_received++;
514
515 // Protocol: A Delay_Req is SENT by a slave and RECEIVED by a master.
516 // Existing implementation only handled master response path. Our unit test
517 // calls process_delay_req() while in an Uncalibrated (slave) state to simulate
518 // emitting a Delay_Req locally (to capture T3) before a corresponding Delay_Resp
519 // arrives. To support deterministic test-driven development without adding
520 // separate test-only APIs, we treat calls in SLAVE/UNCALIBRATED as a local
521 // emission event, recording T3 and setting have_delay_req_. This does not
522 // alter actual network behavior for production because production code sends
523 // Delay_Req via send_delay_req_message() and masters still enter the response
524 // branch below. (FM-008 support: ensures offset samples accumulate.)
525 if (port_data_set_.port_state == PortState::Uncalibrated ||
526 port_data_set_.port_state == PortState::Slave) {
527 // Record local transmit timestamp (T3). If rx_timestamp is zero the test
528 // provided, we still store it; path delay calculation remains valid.
529 delay_req_tx_timestamp_ = rx_timestamp;
530 have_delay_req_ = true;
531 return Types::PTPResult<void>::success();
532 }
533
534 // Master handling path: respond to received Delay_Req
535 if (port_data_set_.port_state != PortState::Master) {
536 return Types::PTPResult<void>::success();
537 }
538
539 // Send Delay_Resp message
540 DelayRespMessage response;
541 response.header.setMessageType(MessageType::Delay_Resp);
542 response.header.reserved_version = (response.header.reserved_version & 0xF0) | (port_data_set_.version_number & 0x0F);
543 response.header.messageLength = sizeof(DelayRespMessage);
544 response.header.domainNumber = config_.domain_number;
545 response.header.sequenceId = message.header.sequenceId;
546 response.header.sourcePortIdentity = port_data_set_.port_identity;
547 response.body.receiveTimestamp = rx_timestamp;
548 response.body.requestingPortIdentity = message.header.sourcePortIdentity;
549
550 if (callbacks_.send_delay_resp) {
551 auto result = callbacks_.send_delay_resp(response);
552 if (result == Types::PTPError::Success) {
553 statistics_.delay_resp_messages_sent++;
554 return Types::PTPResult<void>{};
555 }
556 return Types::PTPResult<void>(result);
557 }
558
559 return Types::PTPResult<void>::success();
560}

759 // Check for timeouts
760 auto result = check_timeouts(current_time);
761 if (!result.is_success()) {
762 return result;
763 }
764
765 // Execute state-specific actions
766 auto exec_result = execute_state_actions();
767 if (!exec_result.is_success()) {
768 return exec_result;
769 }
770 // Attempt offset calculation if all timestamps collected
771 if (have_sync_ && have_follow_up_ && have_delay_req_ && have_delay_resp_) {
772 calculate_offset_and_delay();
773 }
774 // Allow BMCA reevaluation on tick in key states to handle external triggers (e.g., forced tie tests)
775 // Run BMCA periodically in Slave/Uncalibrated states to prune expired foreign masters
776 // per IEEE 1588-2019 Section 9.3.2.5, even without new Announce messages
777 // PreMaster needs BMCA to transition to Master after qualification timeout
778 if ((port_data_set_.port_state == PortState::Listening && foreign_master_count_ > 0) ||
779 port_data_set_.port_state == PortState::PreMaster ||
780 port_data_set_.port_state == PortState::Slave ||
781 port_data_set_.port_state == PortState::Uncalibrated) {
782 run_bmca();
783 }
784 // Health heartbeat emission (FM-007): throttle to 1 second
785 const auto one_second = time_interval_for_log_interval(0, 1);
786 if (is_timeout_expired(last_health_emit_time_, current_time, one_second)) {
787 Common::utils::health::emit();
788 last_health_emit_time_ = current_time;
789 }
790 return Types::PTPResult<void>::success();

937Types::PTPResult<void> PtpPort::run_bmca() noexcept {
938 using namespace IEEE::_1588::PTP::_2019::BMCA;
939 // Minimal BMCA integration (increment 1):
940 // - Build local and foreign priority vectors
941 // - Select best via canonical comparator
942 // - Emit RS_MASTER when local is best and port is Listening
943 // - Else emit RS_SLAVE when foreign is best and port is Listening
944
945 // Prune expired foreign masters before BMCA execution per IEEE 1588-2019 Section 9.3.2.5
946 // This ensures only valid foreign masters participate in Best Master Clock Algorithm
947 Types::Timestamp current_time = callbacks_.get_timestamp ? callbacks_.get_timestamp() : Types::Timestamp{};
948 auto prune_result = prune_expired_foreign_masters(current_time);
949 if (!prune_result.is_success()) {
950 Common::utils::logging::warn("BMCA", 0x0111, "Failed to prune expired foreign masters");
951 }
952
953 // Build local priority vector from LOCAL clock's own parameters (not parent_data_set)
954 // parent_data_set represents the CURRENT MASTER, not the local clock itself
955 // Per IEEE 1588-2019 Section 9.3, BMCA compares local clock parameters to foreign masters
956 // Read from defaultDS per IEEE 1588-2019 Section 8.2.1 (CAP-20251111-01 - Integration Fix)
957 PriorityVector local{};
958 local.priority1 = default_data_set_.priority1; // From defaultDS (not hardcoded 128)
959 local.clockClass = default_data_set_.clockQuality.clock_class; // From defaultDS
960 local.clockAccuracy = default_data_set_.clockQuality.clock_accuracy; // From defaultDS
961 local.variance = default_data_set_.clockQuality.offset_scaled_log_variance; // From defaultDS
962 local.priority2 = default_data_set_.priority2; // From defaultDS (not hardcoded 128)
963 // Use LOCAL port's clock identity for BMCA comparison (not parent_data_set GM identity)
964 // Collapse 8-byte ClockIdentity into u64 for comparator (implementation detail)
965 // Note: This is a simplified monotonic mapping for increment 1; full comparator uses byte-wise ordering.
966 std::uint64_t local_gid = 0;
967 for (int i = 0; i < 8; ++i) {
968 local_gid = (local_gid << 8) | port_data_set_.port_identity.clock_identity[i];
969 }
970 local.grandmasterIdentity = local_gid;
971 local.stepsRemoved = 0; // Local clock is root, always 0 steps removed
972
973 // Assemble list: index 0 = local, followed by foreign entries
974 // Use fixed-size small array then copy into vector only if foreign present (avoid heap unless needed)
975 std::vector<PriorityVector> list; // Phase 05 allowance: vector until replaced with static array wrapper
976 list.reserve(static_cast<size_t>(foreign_master_count_) + 1);
977 list.emplace_back(local);
978 for (std::uint8_t i = 0; i < foreign_master_count_; ++i) {
979 const auto& f = foreign_masters_[i];
980 PriorityVector v{};
981 v.priority1 = f.body.grandmasterPriority1;
982 v.clockClass = f.body.grandmasterClockClass;
983 v.clockAccuracy = f.body.grandmasterClockAccuracy;
984 v.variance = f.body.grandmasterClockVariance;
985 v.priority2 = f.body.grandmasterPriority2;
986 std::uint64_t gid = 0;
987 for (int b = 0; b < 8; ++b) { gid = (gid << 8) | f.body.grandmasterIdentity[b]; }
988 if (gid == 0) {
989 // Fallback: if GM identity not provided, derive from source port clock identity
990 gid = 0;
991 for (int b = 0; b < 8; ++b) { gid = (gid << 8) | f.header.sourcePortIdentity.clock_identity[b]; }
992 }
993 v.grandmasterIdentity = gid;
994 // stepsRemoved is network-order in message body; stored as host in our struct? Convert conservatively.
995 v.stepsRemoved = static_cast<std::uint16_t>(f.body.stepsRemoved);
996 list.emplace_back(v);
997 }
998
999 const int best = selectBestIndex(list);
1000 if (best < 0) {
1001 return Types::PTPResult<void>::success();
1002 }
1003
1004 // Forced tie passive logic (REQ-F-202): if a forced tie was observed this cycle
1005 // recommend PASSIVE regardless of current state to ensure deterministic passive outcome.
1006 bool forced_tie_cycle = Common::utils::fi::was_bmca_tie_forced_and_clear();
1007 if (forced_tie_cycle) {
1008 Common::utils::logging::info("BMCA", 0x0110, "Forced tie detected in run_bmca - issuing PASSIVE recommendation");
1009 Common::utils::metrics::increment(Common::utils::metrics::CounterId::BMCA_PassiveWins, 1);
1010 return process_event(StateEvent::RS_PASSIVE);
1011 }
1012
1013 // BMCA can make state recommendations and update parent_data_set in multiple states
1014 // per IEEE 1588-2019 Section 9.2 - state machine allows transitions based on BMCA results
1015 // Include SLAVE and UNCALIBRATED to allow parent_data_set updates when foreign master changes
1016 if (port_data_set_.port_state == PortState::Listening ||
1017 port_data_set_.port_state == PortState::PreMaster ||
1018 port_data_set_.port_state == PortState::Master ||
1019 port_data_set_.port_state == PortState::Passive ||
1020 port_data_set_.port_state == PortState::Uncalibrated ||
1021 port_data_set_.port_state == PortState::Slave) {
1022
1023 // Special case: If in SLAVE or UNCALIBRATED and all foreign masters pruned (timeout),
1024 // emit ANNOUNCE_RECEIPT_TIMEOUT to transition to Listening per IEEE 1588-2019 Section 9.2.6.11
1025 if ((port_data_set_.port_state == PortState::Slave ||
1026 port_data_set_.port_state == PortState::Uncalibrated) &&
1027 foreign_master_count_ == 0) {
1028 Common::utils::logging::info("BMCA", 0x0112, "No foreign masters after pruning - announce timeout");
1029 return process_event(StateEvent::ANNOUNCE_RECEIPT_TIMEOUT);
1030 }
1031
1032 // Tie handling logic:
1033 // A true tie occurs only if at least one FOREIGN candidate has identical priority vector
1034 // to the LOCAL candidate. Prior code incorrectly treated self-comparison (best==0) as tie.
1035 bool foreign_tie_with_local = false;
1036 if (best == 0) {
1037 // Local selected; check if any foreign vector equals local
1038 for (size_t i = 1; i < list.size(); ++i) {
1039 if (comparePriorityVectors(list[i], list[0]) == CompareResult::Equal) {
1040 foreign_tie_with_local = true;
1041 break;
1042 }
1043 }
1044 if (foreign_tie_with_local) {
1045 Common::utils::metrics::increment(Common::utils::metrics::CounterId::BMCA_PassiveWins, 1);
1046 return process_event(StateEvent::RS_PASSIVE);
1047 }
1048 // Local strictly better master path
1049 // Reset parentDS to local clock values per IEEE 1588-2019 Section 8.2.3
1050 parent_data_set_.grandmaster_identity = port_data_set_.port_identity.clock_identity; // local becomes GM
1051 parent_data_set_.grandmaster_priority1 = 128; // default local priority1
1052 parent_data_set_.grandmaster_priority2 = 128; // default local priority2
1053 parent_data_set_.grandmaster_clock_quality.clock_class = 248; // default slave-only
1054 parent_data_set_.grandmaster_clock_quality.clock_accuracy = 0xFE; // unknown accuracy
1055 parent_data_set_.grandmaster_clock_quality.offset_scaled_log_variance = 0xFFFF; // max variance
1056 parent_data_set_.parent_port_identity = port_data_set_.port_identity; // self as parent
1057 current_data_set_.steps_removed = 0; // root of sync tree
1058 Common::utils::metrics::increment(Common::utils::metrics::CounterId::BMCA_LocalWins, 1);
1059 return process_event(StateEvent::RS_MASTER);
1060 } else {
1061 // Foreign selected; verify if it's an exact tie with local passive; else slave path
1062 if (comparePriorityVectors(list[best], list[0]) == CompareResult::Equal) {
1063 Common::utils::metrics::increment(Common::utils::metrics::CounterId::BMCA_PassiveWins, 1);
1064 return process_event(StateEvent::RS_PASSIVE);
1065 }
1066 const auto &f = foreign_masters_[static_cast<size_t>(best - 1)];
1067 // Assign GM identity; if empty in message, fallback to source port identity
1068 bool gm_zero = true; for (auto b : f.body.grandmasterIdentity) { if (b != 0) { gm_zero = false; break; } }
1069 parent_data_set_.grandmaster_identity = gm_zero ? f.header.sourcePortIdentity.clock_identity
1070 : f.body.grandmasterIdentity;
1071 parent_data_set_.grandmaster_priority1 = f.body.grandmasterPriority1;
1072 parent_data_set_.grandmaster_priority2 = f.body.grandmasterPriority2;
1073 parent_data_set_.grandmaster_clock_quality.clock_class = f.body.grandmasterClockClass;
1074 parent_data_set_.grandmaster_clock_quality.clock_accuracy = f.body.grandmasterClockAccuracy;
1075 parent_data_set_.grandmaster_clock_quality.offset_scaled_log_variance = f.body.grandmasterClockVariance;
1076 parent_data_set_.parent_port_identity = f.header.sourcePortIdentity;
1077 current_data_set_.steps_removed = static_cast<std::uint16_t>(f.body.stepsRemoved + 1); // one additional hop to local
1078 Common::utils::metrics::increment(Common::utils::metrics::CounterId::BMCA_ForeignWins, 1);
1079 return process_event(StateEvent::RS_SLAVE);
1080 }
1081 }
1082
1083 return Types::PTPResult<void>::success();
1084}

1136Types::PTPResult<void> PtpPort::prune_expired_foreign_masters(const Types::Timestamp& current_time) noexcept {
1137 std::uint8_t write_index = 0;
1138 std::uint8_t removed_count = 0;
1139
1140 // Iterate through all foreign masters, keeping only non-expired ones
1141 for (std::uint8_t read_index = 0; read_index < foreign_master_count_; ++read_index) {
1142 const auto& foreign_msg = foreign_masters_[read_index];
1143 const auto& foreign_timestamp = foreign_master_timestamps_[read_index];
1144
1145 // Calculate timeout interval for this foreign master
1146 // Per IEEE 1588-2019 Section 8.2.15.4: timeout = announceReceiptTimeout 2^logMessageInterval
1147 // Use logMessageInterval from the Announce message itself
1148 std::uint8_t log_interval = foreign_msg.header.logMessageInterval;
1149 Types::TimeInterval timeout_interval = time_interval_for_log_interval(
1150 log_interval,
1151 config_.announce_receipt_timeout
1152 );
1153
1154 // Check if this foreign master has expired
1155 bool expired = is_timeout_expired(foreign_timestamp, current_time, timeout_interval);
1156
1157 if (expired) {
1158 // Foreign master expired - skip it (don't copy to write position)
1159 removed_count++;
1160
1161 // Log expiration for debugging
1162 std::uint64_t clock_id = 0;
1163 for (int i = 0; i < 8; ++i) {
1164 clock_id = (clock_id << 8) | foreign_msg.header.sourcePortIdentity.clock_identity[i];
1165 }
1166 Common::utils::logging::info("ForeignMasterList", 0x0302,
1167 "Pruned expired foreign master (timeout exceeded)");
1168 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1169 } else {
1170 // Foreign master still valid - keep it
1171 if (write_index != read_index) {
1172 // Compact: move to earlier position
1173 foreign_masters_[write_index] = foreign_masters_[read_index];
1174 foreign_master_timestamps_[write_index] = foreign_master_timestamps_[read_index];
1175 }
1176 write_index++;
1177 }
1178 }
1179
1180 // Update count to reflect removed entries
1181 foreign_master_count_ = write_index;
1182
1183 // Emit health status if any foreign masters were pruned
1184 if (removed_count > 0) {
1185 Common::utils::health::emit();
1186 }
1187
1188 return Types::PTPResult<void>::success();

1191Types::PTPResult<void> PtpPort::calculate_offset_and_delay() noexcept {
1192 // IEEE 1588-2019 Section 11.1 - Delay mechanism isolation
1193 // In P2P mode, only peer delay mechanism should update mean_path_delay
1194 // E2E calculations are skipped
1195 if (config_.delay_mechanism_p2p) {
1196 // Reset E2E flags but don't calculate - use peer delay instead
1197 have_sync_ = have_follow_up_ = have_delay_req_ = have_delay_resp_ = false;
1198 return Types::PTPResult<void>::success();
1199 }
1200
1201 if (!(have_sync_ && have_follow_up_ && have_delay_req_ && have_delay_resp_)) {
1202 return Types::PTPResult<void>::failure(Types::PTPError::Invalid_Parameter);
1203 }
1204 // Ordering checks (FM-001): warn/telemetry if violated
1205 if (sync_rx_timestamp_ < sync_origin_timestamp_) {
1206 Common::utils::logging::warn("Timestamps", 0x0206, "Sync RX earlier than origin in port calc (T2 < T1)");
1207 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1208 }
1209 if (delay_resp_rx_timestamp_ < delay_req_tx_timestamp_) {
1210 Common::utils::logging::warn("Timestamps", 0x0207, "DelayResp RX earlier than DelayReq TX in port calc (T4 < T3)");
1211 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1212 }
1213 Types::TimeInterval t2_minus_t1 = sync_rx_timestamp_ - sync_origin_timestamp_;
1214 Types::TimeInterval t4_minus_t3 = delay_resp_rx_timestamp_ - delay_req_tx_timestamp_;
1215 double t2_t1_ns = t2_minus_t1.toNanoseconds();
1216 double t4_t3_ns = t4_minus_t3.toNanoseconds();
1217
1218 // Apply correctionField per IEEE 1588-2019 Section 11.3.2
1219 // Total correction = sum of corrections from Sync, Follow_Up, and Delay_Resp messages
1220 double total_correction_ns = sync_correction_.toNanoseconds() +
1221 follow_up_correction_.toNanoseconds() +
1222 delay_resp_correction_.toNanoseconds();
1223
1224 double offset_ns = (t2_t1_ns - t4_t3_ns) / 2.0 + total_correction_ns;
1225 double path_ns = (t2_t1_ns + t4_t3_ns) / 2.0;
1226 // Store only if computed path delay positive (basic validation)
1227 if (path_ns > 0.0) {
1228 current_data_set_.offset_from_master = Types::TimeInterval::fromNanoseconds(offset_ns);
1229 current_data_set_.mean_path_delay = Types::TimeInterval::fromNanoseconds(path_ns);
1230 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsPassed, 1);
1231 Common::utils::metrics::increment(Common::utils::metrics::CounterId::OffsetsComputed, 1); // ensure heuristic sees this sample
1232 Common::utils::health::record_offset_ns(static_cast<long long>(current_data_set_.offset_from_master.toNanoseconds()));
1233 if (port_data_set_.port_state == PortState::Uncalibrated) {
1234 ++successful_offsets_in_window_;
1235 }
1236 // Reset sample flags so next offset requires fresh T1..T4 (prevents double counting per cycle)
1237 have_sync_ = have_follow_up_ = have_delay_req_ = have_delay_resp_ = false;
1238 } else {
1239 Common::utils::logging::warn("Offset", 0x0208, "Computed mean path delay non-positive; values not updated");
1240 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1241 // Reset anyway to force new full sample acquisition
1242 have_sync_ = have_follow_up_ = have_delay_req_ = have_delay_resp_ = false;
1243 }
1244 return Types::PTPResult<void>::success();

1247Types::PTPResult<void> PtpPort::calculate_peer_delay() noexcept {
1248 // IEEE 1588-2019 Section 11.4.2 - Peer delay mechanism
1249 // Formula: <meanPathDelay> = ((t4 - t1) - (t3 - t2) + correctionField) / 2
1250 // where:
1251 // t1 = local transmit timestamp of Pdelay_Req
1252 // t2 = peer receive timestamp of Pdelay_Req (from Pdelay_Resp)
1253 // t3 = peer transmit timestamp of Pdelay_Resp (from Pdelay_Resp_Follow_Up)
1254 // t4 = local receive timestamp of Pdelay_Resp
1255
1256 if (!(have_pdelay_req_ && have_pdelay_resp_ && have_pdelay_resp_follow_up_)) {
1257 return Types::PTPResult<void>::failure(Types::PTPError::Invalid_Parameter);
1258 }
1259
1260 // Ordering validation checks per IEEE specification
1261 if (pdelay_resp_rx_timestamp_ < pdelay_req_tx_timestamp_) {
1262 Common::utils::logging::warn("PeerDelay", 0x0300, "Pdelay_Resp RX earlier than Pdelay_Req TX (t4 < t1)");
1263 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1264 }
1265 if (pdelay_resp_tx_timestamp_ < pdelay_req_rx_timestamp_) {
1266 Common::utils::logging::warn("PeerDelay", 0x0301, "Pdelay_Resp TX earlier than Pdelay_Req RX (t3 < t2)");
1267 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1268 }
1269
1270 // Calculate time intervals
1271 Types::TimeInterval t4_minus_t1 = pdelay_resp_rx_timestamp_ - pdelay_req_tx_timestamp_;
1272 Types::TimeInterval t3_minus_t2 = pdelay_resp_tx_timestamp_ - pdelay_req_rx_timestamp_;
1273 double t4_t1_ns = t4_minus_t1.toNanoseconds();
1274 double t3_t2_ns = t3_minus_t2.toNanoseconds();
1275
1276 // Apply correctionField per IEEE 1588-2019 Section 11.4.2
1277 // Total correction = sum of corrections from Pdelay_Resp and Pdelay_Resp_Follow_Up messages
1278 double total_correction_ns = pdelay_resp_correction_.toNanoseconds() +
1279 pdelay_resp_follow_up_correction_.toNanoseconds();
1280
1281 // Calculate mean path delay
1282 double peer_delay_ns = ((t4_t1_ns - t3_t2_ns) + total_correction_ns) / 2.0;
1283
1284 // Validation: peer delay must be non-negative
1285 if (peer_delay_ns < 0.0) {
1286 Common::utils::logging::warn("PeerDelay", 0x0302, "Computed peer delay negative; measurement rejected");
1287 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsFailed, 1);
1288 // Reset flags to force new measurement
1289 have_pdelay_req_ = have_pdelay_resp_ = have_pdelay_resp_follow_up_ = false;
1290 return Types::PTPResult<void>::success();
1291 }
1292
1293 // Store peer delay result
1294 current_data_set_.mean_path_delay = Types::TimeInterval::fromNanoseconds(peer_delay_ns);
1295 Common::utils::metrics::increment(Common::utils::metrics::CounterId::ValidationsPassed, 1);
1296 Common::utils::logging::debug("PeerDelay", 0x0303, "Peer delay computed successfully");
1297
1298 // Reset flags for next measurement cycle
1299 have_pdelay_req_ = have_pdelay_resp_ = have_pdelay_resp_follow_up_ = false;
1300
1301 return Types::PTPResult<void>::success();
1302}
