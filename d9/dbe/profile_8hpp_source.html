192 ProfileResult validate() const {
193 // Validate domain numbers
194 if (domain_number_max > 127) return ProfileResult::fail();
195 if (domain_number_min > domain_number_max) return ProfileResult::fail();
196
197 // Validate announce receipt timeout
198 if (announce_receipt_timeout < 2) return ProfileResult::fail();
199
200 // Validate message intervals
201 if (announce_interval < -3 || announce_interval > 4) return ProfileResult::fail();
202 if (sync_interval < -7 || sync_interval > 4) return ProfileResult::fail();
203 if (delay_req_interval < -7 || delay_req_interval > 4) return ProfileResult::fail();
204 if (pdelay_req_interval < -7 || pdelay_req_interval > 4) return ProfileResult::fail();
205
206 // Validate delay mechanism consistency
207 if (delay_mechanism == DelayMechanism::E2E) {
208 // E2E should use delay_req, not pdelay_req
209 if (profile_type == PTPProfile::POWER_PROFILE) return ProfileResult::fail(); // Power profile must use P2P
210 } else if (delay_mechanism == DelayMechanism::P2P) {
211 // P2P should use pdelay_req, not delay_req
212 if (profile_type == PTPProfile::DEFAULT_PROFILE) return ProfileResult::fail(); // Default profile must use E2E
213 }
214
215 return ProfileResult::ok();
216 }
217};

317inline IEEE::_1588::_2019::ProfileResult apply_profile(IEEE::_1588::PTP::_2019::Clocks::PortConfiguration& port_config,
318 const IEEE::_1588::_2019::ProfileConfiguration& profile) {
319 // Validate profile before applying
320 auto validation_result = profile.validate();
321 if (!validation_result.is_success()) {
322 return validation_result;
323 }
324
325 // Apply delay mechanism
326 port_config.delay_mechanism_p2p = (profile.delay_mechanism == IEEE::_1588::_2019::DelayMechanism::P2P);
327
328 // Apply message intervals
329 port_config.announce_interval = profile.announce_interval;
330 port_config.sync_interval = profile.sync_interval;
331 port_config.delay_req_interval = profile.delay_req_interval;
332 port_config.announce_receipt_timeout = profile.announce_receipt_timeout;
333
334 return IEEE::_1588::_2019::ProfileResult::ok();
335}
