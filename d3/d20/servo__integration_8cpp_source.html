73bool ServoIntegration::adjust(double offset_ns, std::uint64_t current_time_ns) {
74 if (!is_running_) return false;
75
76 // Track timing for holdover detection
77 if (!first_sample_done_) {
78 last_update_time_ns_ = current_time_ns;
79 state_entry_time_ns_ = current_time_ns;
80 first_sample_done_ = true;
81 }
82
83 // Update state machine based on current offset
84 update_state_machine(offset_ns, current_time_ns);
85
86 // Calculate PI controller output (frequency adjustment)
87 double freq_adjustment_ppb = calculate_pi_output(offset_ns);
88
89 // Apply rate limiting for stability
90 if (config_.enable_rate_limiting) {
91 freq_adjustment_ppb = apply_rate_limiting(freq_adjustment_ppb);
92 }
93
94 // Apply anti-windup to prevent integral explosion
95 if (config_.enable_anti_windup) {
96 apply_anti_windup();
97 }
98
99 // Decide: Step (large offset) vs. Slew (small offset)
100 bool should_step = std::abs(offset_ns) > config_.step_threshold_ns;
101
102 if (should_step && health_.state != ServoState::Locked) {
103 // Phase step for large offsets (when not locked)
104 if (callbacks_.adjust_clock) {
105 auto result = callbacks_.adjust_clock(static_cast<std::int64_t>(offset_ns));
106 if (result == IEEE::_1588::PTP::_2019::Types::PTPError::Success) {
107 statistics_.phase_adjustments++;
108 statistics_.last_phase_adjustment_ns = offset_ns;
109 }
110 }
111 } else {
112 // Frequency slew for small offsets or when locked
113 if (callbacks_.adjust_frequency) {
114 auto result = callbacks_.adjust_frequency(freq_adjustment_ppb);
115 if (result == IEEE::_1588::PTP::_2019::Types::PTPError::Success) {
116 statistics_.frequency_adjustments++;
117 statistics_.last_freq_adjustment_ppb = freq_adjustment_ppb;
118 last_freq_adjustment_ppb_ = freq_adjustment_ppb;
119 }
120 }
121 }
122
123 // Update statistics and health
124 update_statistics(offset_ns, freq_adjustment_ppb);
125 update_health_status(current_time_ns);
126
127 // Track timing
128 last_update_time_ns_ = current_time_ns;
129 statistics_.total_adjustments++;
130
131 return true;
132}

138void ServoIntegration::update_state_machine(double offset_ns, std::uint64_t current_time_ns) {
139 double abs_offset = std::abs(offset_ns);
140 ServoState old_state = health_.state;
141 ServoState new_state = old_state;
142
143 // Check for holdover timeout
144 if (config_.enable_holdover && first_sample_done_) {
145 std::uint64_t time_since_update = (current_time_ns - last_update_time_ns_) / 1000000; // ns to ms
146 if (time_since_update > config_.holdover_timeout_ms) {
147 new_state = ServoState::Holdover;
148 }
149 }
150
151 // State transitions based on offset magnitude (unless in holdover)
152 if (new_state != ServoState::Holdover) {
153 if (abs_offset < config_.lock_threshold_ns) {
154 // Within lock threshold
155 consecutive_samples_in_threshold_++;
156
157 if (consecutive_samples_in_threshold_ >= config_.samples_for_lock) {
158 new_state = ServoState::Locked;
159 } else if (old_state != ServoState::Locked) {
160 new_state = ServoState::Locking;
161 }
162 } else if (abs_offset < config_.locking_threshold_ns) {
163 // Within locking threshold
164 consecutive_samples_in_threshold_ = 0;
165 new_state = ServoState::Locking;
166 } else {
167 // Outside thresholds
168 consecutive_samples_in_threshold_ = 0;
169
170 // Only unlock if significantly exceeds threshold
171 if (abs_offset > config_.unlock_threshold_ns) {
172 new_state = ServoState::Unlocked;
173 if (old_state == ServoState::Locked) {
174 statistics_.lock_loss_count++;
175 }
176 }
177 }
178 }
179
180 // Update state if changed
181 if (new_state != old_state) {
182 health_.state = new_state;
183 state_entry_time_ns_ = current_time_ns;
184
185 // Reset integral error on major state changes
186 if (new_state == ServoState::Unlocked || new_state == ServoState::Holdover) {
187 statistics_.integral_error = 0.0;
188 }
189 }
190}

196double ServoIntegration::calculate_pi_output(double offset_ns) {
197 // Proportional term: Kp * error
198 double proportional = config_.kp * offset_ns;
199
200 // Integral term: Ki * integral(error)
201 // Accumulate error (convert ns to equivalent frequency error)
202 // Heuristic: 1ns offset â‰ˆ 1ppb frequency error over 1 second
203 statistics_.integral_error += offset_ns;
204 double integral = config_.ki * statistics_.integral_error;
205
206 // Store terms for statistics
207 statistics_.proportional_term = proportional;
208 statistics_.integral_term = integral;
209
210 // PI output: P + I
211 double output_ppb = proportional + integral;
212
213 // Clamp to maximum frequency adjustment
214 output_ppb = std::max(-config_.max_freq_adjustment_ppb,
215 std::min(config_.max_freq_adjustment_ppb, output_ppb));
216
217 return output_ppb;
218}

224double ServoIntegration::apply_rate_limiting(double requested_ppb) {
225 // Calculate maximum allowed change per sample
226 // Assume 1 sample per second (typical sync interval)
227 double max_change = config_.max_rate_of_change_ppb_per_sec;
228
229 double delta = requested_ppb - last_freq_adjustment_ppb_;
230
231 if (std::abs(delta) > max_change) {
232 // Limit rate of change
233 double limited_ppb = last_freq_adjustment_ppb_ +
234 (delta > 0 ? max_change : -max_change);
235
236 statistics_.rate_limit_hits++;
237 return limited_ppb;
238 }
239
240 return requested_ppb;
241}

279void ServoIntegration::update_health_status(std::uint64_t current_time_ns) {
280 health_.timestamp_ns = current_time_ns;
281
282 // Calculate time since last update
283 if (first_sample_done_) {

286
287 // Calculate time in current state
288 if (first_sample_done_) {
289 health_.time_in_current_state_ms = (current_time_ns - state_entry_time_ns_) / 1000000;
290 }
291
292 // Update health indicators
293 double abs_offset = std::abs(statistics_.last_offset_ns);
294
295 health_.adjustments_active = (statistics_.total_adjustments > 0);
296 health_.phase_locked = (abs_offset < config_.lock_threshold_ns);
297 health_.within_spec = (abs_offset < config_.lock_threshold_ns); // <1Âµs spec
298 health_.frequency_stable = (std::abs(last_freq_adjustment_ppb_) < 10.0); // <10ppb
299
300 // Update status message
301 switch (health_.state) {
302 case ServoState::Uninitialized:
303 health_.message = "Servo not initialized";
304 break;
305 case ServoState::Unlocked:
306 health_.message = "Unlocked - large offset, converging";
307 break;
308 case ServoState::Locking:
309 health_.message = "Locking - offset reducing, approaching lock";
310 break;
311 case ServoState::Locked:

313 break;

315 health_.message = "Holdover - no recent updates, maintaining last frequency";
